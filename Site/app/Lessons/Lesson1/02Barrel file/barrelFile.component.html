Barrel file

Barrel file - это специальный файл который включает несколько експоротов. Он делает реекспорт модулей упрощая доступ к ним
так же барел файл сокращает количество импортов import {RouterModule} from "@angular/router"
итак как же используються барел фафды в данном примере. идея в том чтобы все импорты касающиеся папки lists поместить в один файл с которым потом работать
С баррел файлом и без

без баррел файла

        import { NgModule } from "@angular/core";
        import { BrowserModule } from "@angular/platform-browser";
        import { RouterModule } from "@angular/router"; // модуль для маршрутизации

        import { AppComponent } from "./app.component";                         импорт главной компонены
        import { List1Component } from "./list1/list1.component";               импорт лист 1 компоненты
        import { List2Component } from "./list2/list2.component";               импорт лист 2 компоненты

        @NgModule({
            imports: [
                BrowserModule,
                RouterModule.forRoot([ // настройка маршрутов                             настройка роутинга
                    { path: "list1", component: List1Component }, 
                    { path: "list2", component: List2Component },
                    { path: "", redirectTo: "list1", pathMatch: "full" } 
                ])
            ],
            declarations: [AppComponent, List1Component, List2Component],                  декларация используемых компонент 
            bootstrap: [AppComponent]
        })
        export class AppModule { }

С баррел файлом

            import { NgModule } from "@angular/core";
            import { BrowserModule } from "@angular/platform-browser";
            import { RouterModule, Routes } from "@angular/router"; // модуль для маршрутизации

            import { List1Component, List2Component, List3Component } from "./lists/index";  импорт компонент из баррел файла          
            import { AppComponent } from "./app.component";
            import { ListsModule, routes } from  "./lists/index"; импорт компонент из баррел файла 

            @NgModule({
                imports: [
                    BrowserModule,
                    ListsModule, импорт модуля из баррел файла 
                    RouterModule.forRoot(routes)
                ],
                declarations: [AppComponent],
                bootstrap: [AppComponent]
            })
            export class AppModule { }
